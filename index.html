// Webhook handler: transaction.pending const express = require('express'); const pagoasap = require('@pagoasap/node'); const client = new pagoasap.Client({ clientId: process.env.PAGOASAP_CLIENT_ID, secret: process.env.PAGOASAP_SECRET, }); app.post('/webhook', async (req, res) => { const event = pagoasap.verifyWebhook( req.body, req.headers['x-pagoasap-signature'], process.env.WEBHOOK_SECRET ); if (event.type === 'transaction.pending') { const { id, amount, merchant } = event.data; // Tu lógica de aprobación personalizada const shouldApprove = await checkUserLimits( event.data.cardholder_id, amount ); if (shouldApprove) { await client.transactions.approve(id); } else { await client.transactions.decline(id, { decline_reason: 'spending_limit_exceeded' }); } } res.json({ received: true }); });
